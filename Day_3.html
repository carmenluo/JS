/**
*   Return the second largest number in the array.
*   @param {Number[]} nums - An array of numbers.
*   @return {Number} The second largest number in the array.
**/
function getSecondLargest(nums) {
    // Complete the function
    nums.sort((x, y) => x < y);
    for (let a of nums) {
        if (a < nums[0]) {
            return a;
        }
    }
    
}
